// ===================== TOAST =====================
function toast(title, desc, type='success') {
  const c = document.getElementById('toastContainer');
  const colors = type==='error' ? 'bg-red-50 border-red-200 text-red-700' : type==='warning' ? 'bg-amber-50 border-amber-200 text-amber-700' : 'bg-green-50 border-green-200 text-green-700';
  const el = document.createElement('div');
  el.className = `${colors} border rounded-lg p-3 shadow-lg max-w-sm animate-fade-in`;
  el.innerHTML = `<p class="font-semibold text-sm">${title}</p>${desc ? `<p class="text-xs mt-0.5">${desc}</p>` : ''}`;
  c.appendChild(el);
  setTimeout(() => el.remove(), 3500);
}

// ===================== AUTH =====================
function toggleAuthMode() {
  isRegister = !isRegister;
  document.getElementById('authTitle').textContent = isRegister ? 'Create Account' : 'Sign In';
  document.getElementById('authBtn').textContent = isRegister ? 'Create Account' : 'Sign In';
  document.getElementById('authNameField').classList.toggle('hidden', !isRegister);
  document.getElementById('authToggle').innerHTML = isRegister
    ? 'Already have an account? <button onclick="toggleAuthMode()" class="text-pink-600 font-medium underline">Sign In</button>'
    : 'Don\'t have an account? <button onclick="toggleAuthMode()" class="text-pink-600 font-medium underline">Create Account</button>';
}

async function handleAuth() {
  const email = document.getElementById('authEmail').value.trim();
  const password = document.getElementById('authPassword').value;
  const name = document.getElementById('authName').value.trim();
  const errEl = document.getElementById('authError');
  errEl.classList.add('hidden');

  if (!email || !password) { errEl.textContent='Email and password required'; errEl.classList.remove('hidden'); return; }
  if (password.length < 6) { errEl.textContent='Password must be at least 6 characters'; errEl.classList.remove('hidden'); return; }

  try {
    if (isRegister) {
      const { error } = await sb.auth.signUp({ email, password, data:{ name: name||email.split('@')[0] }});
      if (error) throw error;
      toast('Account Created!', 'You can now sign in.');
      toggleAuthMode();
    } else {
      const { data, error } = await sb.auth.signInWithPassword({ email, password });
      if (error) throw error;
      currentUser = data.user;
      showApp();
    }
  } catch(e) {
    errEl.textContent = e.message || 'Authentication failed';
    errEl.classList.remove('hidden');
  }
}

async function handleSignOut() {
  await sb.auth.signOut();
  currentUser = null;
  dataCache = { customers:[], orders:[], expenses:[], payments:[], notes:[] };
  showAuth();
}

// ===================== DATA LAYER =====================
async function fetchAllData() {
  try {
    const [custRes, orderRes, expRes, payRes, noteRes] = await Promise.all([
      sb.from('customers').select('*').order('created_at',{ascending:true}),
      sb.from('orders').select('*, order_items(*), payments(*)').order('created_at',{ascending:true}),
      sb.from('expenses').select('*').order('date',{ascending:true}),
      sb.from('payments').select('*').order('date',{ascending:true}),
      sb.from('notes').select('*').order('created_at',{ascending:true}),
    ]);
    dataCache.customers = custRes.data || [];
    dataCache.orders = (orderRes.data || []).map(o => ({
      ...o,
      items: o.order_items || [],
      customer: dataCache.customers.find(c=>c.id===o.customer_id) || null
    }));
    // Link customers to orders
    dataCache.orders.forEach(o => {
      o.customer = dataCache.customers.find(c=>c.id===o.customer_id) || null;
    });
    dataCache.expenses = expRes.data || [];
    dataCache.payments = payRes.data || [];
    dataCache.notes = noteRes.data || [];
  } catch(e) {
    console.error('Fetch error:', e);
    toast('Data Error', 'Failed to load data from Supabase', 'error');
  }
}

async function createCustomer(data) {
  const { data: last } = await sb.from('customers').select('customer_id').order('customer_id',{ascending:false}).limit(1);
  let nextNum = 1;
  if (last && last[0]) { const p = parseInt(last[0].customer_id.replace('SK-',''),10); if(!isNaN(p)) nextNum=p+1; }
  const customerId = `SK-${String(nextNum).padStart(4,'0')}`;
  const { data: created, error } = await sb.from('customers').insert({...data, customer_id:customerId}).select().single();
  if (error) throw error;
  return created;
}

async function createOrder(customerData, items) {
  // Find or create customer
  let customer = dataCache.customers.find(c=>c.phone===customerData.phone && c.name===customerData.name);
  if (!customer) {
    customer = await createCustomer({ name:customerData.name, city:customerData.city, phone:customerData.phone, address:customerData.address||'' });
    dataCache.customers.push(customer);
  } else {
    const { data:updated } = await sb.from('customers').update({city:customerData.city,phone:customerData.phone,address:customerData.address||customer.address}).eq('id',customer.id).select().single();
    if(updated) { const i=dataCache.customers.findIndex(c=>c.id===customer.id); if(i>=0) dataCache.customers[i]=updated; customer=updated; }
  }
  // Create order
  const { data:order, error } = await sb.from('orders').insert({customer_id:customer.id,status:'pending'}).select().single();
  if (error) throw error;
  // Create items
  const orderItems = items.map(it=>({...it, order_id:order.id}));
  const { data:createdItems } = await sb.from('order_items').insert(orderItems).select();
  order.items = createdItems || [];
  order.customer = customer;
  order.payments = [];
  dataCache.orders.unshift(order);
  return order;
}

// ===================== HELPERS =====================
function fmt(n) { return `Rs. ${Number(n||0).toLocaleString()}`; }
function fmtDate(d) { return d ? new Date(d).toLocaleDateString('en-PK',{day:'2-digit',month:'short',year:'numeric'}) : ''; }
function fmtDateTime(d) { return d ? new Date(d).toLocaleDateString('en-PK',{day:'2-digit',month:'short',year:'numeric',hour:'2-digit',minute:'2-digit'}) : ''; }
function shortId(id) { return id ? id.slice(0,8).toUpperCase() : ''; }
function getOrderTotal(order) { return (order.items||[]).reduce((s,i)=>s+i.sale_price*i.quantity+i.shipping_charges,0); }
function getOrderCostTotal(order) { return (order.items||[]).reduce((s,i)=>s+i.cost_price*i.quantity,0); }
function getOrderShippingTotal(order) { return (order.items||[]).reduce((s,i)=>s+i.shipping_charges,0); }
function getPaymentsTotal(payments) { return (payments||[]).reduce((s,p)=>s+p.amount,0); }
function escHtml(s) { const d=document.createElement('div'); d.textContent=s; return d.innerHTML; }

function getFilteredOrders() {
  if (dateFilter==='lifetime') return dataCache.orders;
  const now = new Date();
  let start;
  if (dateFilter==='today') { start=new Date(now.getFullYear(),now.getMonth(),now.getDate()); }
  else if (dateFilter==='week') { start=new Date(now); start.setDate(start.getDate()-7); }
  else if (dateFilter==='month') { start=new Date(now.getFullYear(),now.getMonth(),1); }
  else if (dateFilter==='year') { start=new Date(now.getFullYear(),0,1); }
  else if (dateFilter==='date') { start=new Date(filterDate); }
  else return dataCache.orders;
  return dataCache.orders.filter(o=>new Date(o.created_at)>=start);
}

async function submitOrder() {
  const name=document.getElementById('oName').value.trim();

  if(!name||!city||!phone){toast('Validation','Name, city and phone required','error');return;}

  const rows=document.getElementById('productRows').children;
  const items=[];
  for(const row of rows){
    const n=row.querySelector('.prod-name').value.trim();
    if(!n)continue;
    items.push({product_name:n,product_image:row.querySelector('.prod-img').value.trim()||'',quantity:parseInt(row.querySelector('.prod-qty').value)||1,cost_price:parseFloat(row.querySelector('.prod-cost').value)||0,sale_price:parseFloat(row.querySelector('.prod-sale').value)||0,shipping_charges:parseFloat(row.querySelector('.prod-ship').value)||0});

  try{
    await createOrder({name,city,phone,address:addr},items);
    toast('Order Created!',`Order for ${name} created.`);
    productRowCount=0;

  }catch(e){toast('Error',e.message,'error');}

async function collectPayment() {
  const amt=parseFloat(document.getElementById('payAmt').value);
  const note=document.getElementById('payNote').value.trim();
  if(!amt||amt<=0){toast('Validation','Enter a valid amount','error');return;}

  const deliveredOrders=dataCache.orders.filter(o=>o.status==='delivered');
  const totalPending=deliveredOrders.reduce((s,o)=>{const t=getOrderTotal(o);const p=getPaymentsTotal(o.payments);return s+(t-p>0?t-p:0);},0);
  if(amt>totalPending){toast('Amount Exceeds Pending',`Total pending is ${fmt(totalPending)}`,'error');return;}

  // Distribute FIFO across pending orders
  let remaining=amt;
  try{
    for(const o of deliveredOrders){
      if(remaining<=0)break;
      const orderTotal=getOrderTotal(o);
      const collected=getPaymentsTotal(o.payments);
      const pending=orderTotal-collected;
      if(pending<=0)continue;
      const allocate=Math.min(remaining,pending);
      await createPayment({order_id:o.id,amount:allocate,note:note||'Bulk payment collection'});
      remaining-=allocate;
    }
    toast('Payment Collected!',`${fmt(amt)} collected successfully`);
    renderSection();
  }catch(e){toast('Error',e.message,'error');}
}

  try{
    const wb=XLSX.utils.book_new();
    const custData=dataCache.customers.map(c=>({id:c.id,customer_id:c.customer_id,name:c.name,city:c.city,phone:c.phone,address:c.address,created_at:c.created_at,updated_at:c.updated_at}));
    XLSX.utils.book_append_sheet(wb,XLSX.utils.json_to_sheet(custData.length?custData:[{}]),'Customers');

    const orderData=dataCache.orders.map(o=>({id:o.id,customer_id:o.customer_id,status:o.status,invoice_sent:o.invoice_sent,created_at:o.created_at,updated_at:o.updated_at}));
    XLSX.utils.book_append_sheet(wb,XLSX.utils.json_to_sheet(orderData.length?orderData:[{}]),'Orders');

    const itemData=[];
    dataCache.orders.forEach(o=>(o.items||[]).forEach(i=>itemData.push({id:i.id,order_id:i.order_id,product_name:i.product_name,product_image:i.product_image,quantity:i.quantity,cost_price:i.cost_price,sale_price:i.sale_price,shipping_charges:i.shipping_charges})));
    XLSX.utils.book_append_sheet(wb,XLSX.utils.json_to_sheet(itemData.length?itemData:[{}]),'OrderItems');

    const expData=dataCache.expenses.map(e=>({id:e.id,category:e.category,amount:e.amount,description:e.description,date:e.date,created_at:e.created_at,updated_at:e.updated_at}));
    XLSX.utils.book_append_sheet(wb,XLSX.utils.json_to_sheet(expData.length?expData:[{}]),'Expenses');

    const payData=dataCache.payments.map(p=>({id:p.id,order_id:p.order_id,amount:p.amount,date:p.date,note:p.note,created_at:p.created_at,updated_at:p.updated_at}));
    XLSX.utils.book_append_sheet(wb,XLSX.utils.json_to_sheet(payData.length?payData:[{}]),'Payments');

    const noteData=dataCache.notes.map(n=>({id:n.id,title:n.title,content:n.content,created_at:n.created_at,updated_at:n.updated_at}));
    XLSX.utils.book_append_sheet(wb,XLSX.utils.json_to_sheet(noteData.length?noteData:[{}]),'Notes');

    XLSX.writeFile(wb,`stylo-kidz-backup-${new Date().toISOString().replace(/[:.]/g,'-').slice(0,19)}.xlsx`);
    toast('Export Complete!','All data exported as XLSX');
  }catch(e){toast('Export Failed',e.message,'error');}
}

async function importXLSX() {
  const fileInput=document.getElementById('importFile');
  if(!fileInput.files[0]){toast('No File','Select an XLSX file','error');return;}
  try{
    const buf=await fileInput.files[0].arrayBuffer();
    const wb=XLSX.read(buf,{type:'array'});
    const result={customers:0,orders:0,orderItems:0,expenses:0,payments:0,notes:0};

    // Import Customers
    if(wb.SheetNames.includes('Customers')){
      const rows=XLSX.utils.sheet_to_json(wb.Sheets['Customers']);
      for(const r of rows){
        if(!r.name)continue;
        try{const existing=dataCache.customers.find(c=>c.id===r.id);
        if(existing){await sb.from('customers').update({customer_id:r.customer_id,name:r.name,city:r.city||'',phone:r.phone||'',address:r.address||''}).eq('id',r.id);}
        else{await sb.from('customers').insert({id:r.id,customer_id:r.customer_id,name:r.name,city:r.city||'',phone:r.phone||'',address:r.address||'',created_at:r.created_at,updated_at:r.updated_at});}
        result.customers++;}catch(e){console.error('Import customer:',e);}
      }
    }

    // Import Orders
    if(wb.SheetNames.includes('Orders')){
      const rows=XLSX.utils.sheet_to_json(wb.Sheets['Orders']);
      for(const r of rows){
        if(!r.customer_id)continue;
        try{const existing=dataCache.orders.find(o=>o.id===r.id);
        if(existing){await sb.from('orders').update({status:r.status||'pending',invoice_sent:r.invoice_sent||false}).eq('id',r.id);}
        else{await sb.from('orders').insert({id:r.id,customer_id:r.customer_id,status:r.status||'pending',invoice_sent:r.invoice_sent||false,created_at:r.created_at,updated_at:r.updated_at});}
        result.orders++;}catch(e){console.error('Import order:',e);}
      }
    }

    // Import OrderItems
    if(wb.SheetNames.includes('OrderItems')){
      const rows=XLSX.utils.sheet_to_json(wb.Sheets['OrderItems']);
      for(const r of rows){
        if(!r.order_id||!r.product_name)continue;
        try{await sb.from('order_items').insert({id:r.id,order_id:r.order_id,product_name:r.product_name,product_image:r.product_image||'',quantity:r.quantity||1,cost_price:r.cost_price||0,sale_price:r.sale_price||0,shipping_charges:r.shipping_charges||0});
        result.orderItems++;}catch(e){console.error('Import item:',e);}
      }
    }

    // Import Expenses
    if(wb.SheetNames.includes('Expenses')){
      const rows=XLSX.utils.sheet_to_json(wb.Sheets['Expenses']);
      for(const r of rows){
        if(!r.category&&!r.amount)continue;
        try{await sb.from('expenses').insert({id:r.id,category:r.category||'Others',amount:r.amount||0,description:r.description||'',date:r.date||new Date().toISOString(),created_at:r.created_at,updated_at:r.updated_at});
        result.expenses++;}catch(e){console.error('Import expense:',e);}
      }
    }

    // Import Payments
    if(wb.SheetNames.includes('Payments')){
      const rows=XLSX.utils.sheet_to_json(wb.Sheets['Payments']);
      for(const r of rows){
        if(!r.order_id||!r.amount)continue;
        try{await sb.from('payments').insert({id:r.id,order_id:r.order_id,amount:r.amount,note:r.note||'',date:r.date||new Date().toISOString(),created_at:r.created_at,updated_at:r.updated_at});
        result.payments++;}catch(e){console.error('Import payment:',e);}
      }
    }

    // Import Notes
    if(wb.SheetNames.includes('Notes')){
      const rows=XLSX.utils.sheet_to_json(wb.Sheets['Notes']);
      for(const r of rows){
        if(!r.title)continue;
        try{await sb.from('notes').insert({id:r.id,title:r.title,content:r.content||'',created_at:r.created_at,updated_at:r.updated_at});
        result.notes++;}catch(e){console.error('Import note:',e);}
      }
    }

    await fetchAllData();
    const total=result.customers+result.orders+result.orderItems+result.expenses+result.payments+result.notes;
    document.getElementById('importResult').innerHTML=`<div class="mt-3 bg-green-50 border border-green-200 rounded-lg p-3 text-sm text-green-700">✅ Imported: ${result.customers} customers, ${result.orders} orders, ${result.orderItems} items, ${result.expenses} expenses, ${result.payments} payments, ${result.notes} notes</div>`;
    toast('Import Complete!',`${total} records imported`);
  }catch(e){toast('Import Failed',e.message,'error');}
}

async function submitNote() {

}

async function updateName() {

async function checkAndInit() {
  // Check if tables exist by trying to query customers
  try {
    const { error } = await sb.from('customers').select('id').limit(1);
    if (error && error.code === '42P01') {
      // Tables don't exist
      showSetup();
      return;
    }
    // Tables exist, check auth
    const { data: { session } } = await sb.auth.getSession();
    if (session) {
      currentUser = session.user;
      showApp();
    } else {
      showAuth();
    }
  } catch(e) {
    console.error('Init error:', e);
    // Might be network error, try showing auth
    const { data: { session } } = await sb.auth.getSession();
    if (session) {
      currentUser = session.user;
      showApp();
    } else {
      showAuth();
    }
  }
}
